/**
 * AI分析结果值对象
 */
@Data
public class AiAnalysisResultVo {
    /**
     * AI生成可能性分数（AI生成痕迹明显程度的数值，百分制，0-100，分数越高说明由AI生成的痕迹越明显）
     * 注意：此字段由系统通过数学公式自动计算，你不需要返回此字段
     */
    private Integer aiScore;

    /**
     * AI痕迹分析维度（基于数学公式计算，共6个维度）
     * 注意：此字段由系统通过数学公式自动计算，你不需要返回此字段
     */
    private List<AiDimensionVo> aiDimensions;

    /**
     * 质量维度
     * 维度名有六个：志望動機の明確性と具体性、学習計画と未来目標の合理性、表現力と説得力、学部専門との適合度、文章構造と論理展開、文法と日本語の正確性
     * 每个QualityDimensionVo对象包括：维度名、该维度下的优秀度打分和该维度下的评语说明（2句话的篇幅）
     */
    private List<QualityDimensionVo> qualityDimensions;

    /**
     * 修改建议
     * 首先给出完整的修改建议，然后给出修改后的完整材料
     */
    private List<String> recommendations;
}

/**
 * AI生成痕迹分析维度值对象
 * 注意：此对象由系统通过数学公式自动生成，你不需要返回此对象
 */
@Data
public class AiDimensionVo {
    /**
     * 维度名称
     */
    private String name;

    /**
     * 级别 (0.0-10.0，保留1位小数，分数越高越接近人类写作)
     */
    private Double level;

    /**
     * 评价
     * 大约2句话的文字篇幅量
     */
    private String evaluation;
}

/**
 * 质量维度值对象
 */
@Data
public class QualityDimensionVo {
    /**
     * 维度名称
     */
    private String name;

    /**
     * 分数 (0-100)
     */
    private Integer score;

    /**
     * 评价
     * 大约2句话的篇幅量
     */
    private String evaluation;
}
